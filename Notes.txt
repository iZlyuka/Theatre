void TicketsControl::printTickets() {
	setlocale(0, "rus");
	ifstream file("Tickets.txt");
	if (!file.is_open()) {
		cout << "Спектаклей нет!" << '\n';
		return;
	}
	string line;
	Head();
	for (int i = 0; !file.eof(); i++) {
		getline(file, line);
		if (line.empty()) {
			continue;
		}
		cout << i << ' ' << line << endl;
	}
	file.close();
}

	vector<string> test;

	string ticket;
	string* FTickets = nullptr;
	string* tmp = nullptr;
	int k = 0;
		while (!file.eof()) {
			test.push_back(ticket);
			f = find(test.begin(), test.end(), STicket);
			if (f != test.end()) {
				cout << f - test.begin();
				return f - test.begin();
				test2.push_back(f - test.begin());
		}
	}
	while (!file.eof()) {
		getline(file, ticket);
		if (ticket.find(STicket) != -1) {
			k++;
			tmp = new string[k];
			for (int i = 0; i < k - 1; i++) {
				tmp[i] = FTickets[i];
			}
			tmp[k - 1] = ticket;
			delete[] FTickets;
			FTickets = tmp;
		}
	}
	file.close();
	count = k;
	return FTickets;
}
